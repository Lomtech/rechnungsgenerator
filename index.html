🔧 Google Sheet Integration einrichten

📋 Schritt 1: Google Sheet erstellen
  1. Gehen Sie zu sheets.google.com
  2. Erstellen Sie ein neues Sheet (z.B. "Rechnungen")
  3. Tragen Sie in der ersten Zeile diese Spaltenüberschriften ein:
A1: Typ | B1: Rechnungsnummer | C1: Firma | D1: Kunde | E1: Anrede
F1: Kundennummer | G1: Rechnungsdatum | H1: Leistungsdatum | I1: Position | J1: Nettosumme | K1: MwSt-Satz | L1: MwSt-Betrag | M1: Gesamtbetrag
⚙️ Schritt 2: Apps Script erstellen
  1. Im Google Sheet: Erweiterungen → Apps Script
  2. Löschen Sie alles im Editor und fügen Sie diesen Code ein: 
function doGet(e) {
  // Prüfen ob e und parameter existieren
  if (e && e.parameter) {
    if (e.parameter.action === 'getNextNumber') {
      return getNextInvoiceNumber(e.parameter.type);
    }
    if (e.parameter.action === 'getNextCustomerNumber') {
      return getNextCustomerNumber();
    }
    if (e.parameter.action === 'searchCustomer') {
      return searchExistingCustomer(e.parameter.name);
    }
  }
  
  return ContentService.createTextOutput("Script läuft!");
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  try {
    var data = JSON.parse(e.postData.contents);
    
    sheet.appendRow([
      data.type || "",                    // A: Typ
      data.invoiceNumber || "",           // B: Rechnungsnummer  
      data.company || "",                 // C: Firma
      data.client || "",                  // D: Kunde
      data.clientGender || "",            // E: Anrede
      data.customerNumber || "",          // F: Kundennummer
      data.invoiceDate || "",             // G: Rechnungsdatum
      data.serviceDate || "",             // H: Leistungsdatum
      data.items || "",                   // I: Position
      data.netTotal || "",                // J: Nettosumme
      data.taxRate + "%" || "",           // K: MwSt-Satz
      data.taxAmount || "",               // L: MwSt-Betrag
      data.grandTotal || ""               // M: Gesamtbetrag
    ]);
    
  } catch (error) {
    sheet.appendRow(["FEHLER", error.toString()]);
  }
  
  return ContentService.createTextOutput("Daten gespeichert");
}

function searchExistingCustomer(customerName, customerAddress) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = sheet.getDataRange().getValues();
    
    // Normalisiere Suchkriterien
    var searchName = customerName.toLowerCase().replace(/\s+/g, '');
    var searchAddress = customerAddress ? customerAddress.toLowerCase().replace(/\s+/g, '') : '';
    
    var bestMatch = null;
    var bestScore = 0;
    
    // Durchsuche alle Zeilen nach dem Kundennamen und Adresse
    for (var i = 1; i < data.length; i++) {
      var existingClient = data[i][3]; // Spalte D (Kunde)
      var existingGender = data[i][4]; // Spalte E (Anrede)
      var existingCustomerNumber = data[i][5]; // Spalte F (Kundennummer)
      var existingAddress = data[i][7] || ''; // Spalte H (Adresse) - falls verfügbar
      
      if (existingClient && typeof existingClient === 'string') {
        var existingName = existingClient.toLowerCase().replace(/\s+/g, '');
        var existingAddr = existingAddress.toString().toLowerCase().replace(/\s+/g, '');
        
        var score = 0;
        
        // Name-Matching (Hauptkriterium)
        if (existingName === searchName) {
          score += 100; // Exakte Übereinstimmung
        } else if (existingName.includes(searchName) || searchName.includes(existingName)) {
          score += 60; // Teilweise Übereinstimmung
        } else {
          // Prüfe einzelne Wörter
          var nameWords = searchName.split(/[^a-z0-9]/);
          var existingWords = existingName.split(/[^a-z0-9]/);
          var matchingWords = 0;
          
          nameWords.forEach(function(word) {
            if (word.length > 2 && existingWords.some(function(eWord) { return eWord.includes(word); })) {
              matchingWords++;
            }
          });
          
          if (matchingWords >= 2) {
            score += 40; // Mehrere Wörter stimmen überein
          } else if (matchingWords >= 1) {
            score += 20; // Ein Wort stimmt überein
          }
        }
        
        // Adress-Matching (zusätzliches Kriterium)
        if (searchAddress && existingAddr) {
          if (existingAddr.includes(searchAddress) || searchAddress.includes(existingAddr)) {
            score += 30; // Adresse stimmt überein
          } else {
            // Prüfe Straßenname oder PLZ
            var addressParts = searchAddress.split(/[^a-z0-9]/);
            var existingParts = existingAddr.split(/[^a-z0-9]/);
            var addressMatches = 0;
            
            addressParts.forEach(function(part) {
              if (part.length > 2 && existingParts.some(function(ePart) { return ePart.includes(part); })) {
                addressMatches++;
              }
            });
            
            if (addressMatches >= 1) {
              score += 15; // Teilweise Adress-Übereinstimmung
            }
          }
        }
        
        // Speichere den besten Match
        if (score > bestScore && score >= 60 && existingCustomerNumber) { // Mindest-Score für Match
          bestScore = score;
          bestMatch = {
            customerNumber: existingCustomerNumber,
            gender: existingGender || '',
            score: score
          };
        }
      }
    }
    
    // Gib den besten Match zurück
    if (bestMatch) {
      var result = bestMatch.customerNumber;
      if (bestMatch.gender) {
        result += "|" + bestMatch.gender;
      }
      result += "|SCORE:" + bestMatch.score; // Für Debugging
      return ContentService.createTextOutput(result);
    }
    
    return ContentService.createTextOutput("NICHT_GEFUNDEN");
    
  } catch (error) {
    return ContentService.createTextOutput("FEHLER: " + error.toString());
  }
}

function getNextInvoiceNumber(type) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = sheet.getDataRange().getValues();
    
    var maxNumber = 0;
    var prefix = type === 'Angebot' ? 'ANG-' : 'RE-';
    
    for (var i = 1; i < data.length; i++) {
      var invoiceNumber = data[i][1]; // Spalte B (Rechnungsnummer)
      
      if (invoiceNumber && typeof invoiceNumber === 'string') {
        var numberPart = invoiceNumber.replace(prefix, '').replace(/[^0-9]/g, '');
        
        if (numberPart) {
          var num = parseInt(numberPart);
          if (!isNaN(num) && num > maxNumber) {
            maxNumber = num;
          }
        }
      }
    }
    
    var nextNumber = maxNumber + 1;
    var paddedNumber = nextNumber.toString().padStart(4, '0');
    var result = prefix + paddedNumber;
    
    return ContentService.createTextOutput(result);
    
  } catch (error) {
    return ContentService.createTextOutput("FEHLER: " + error.toString());
  }
}

function getNextCustomerNumber() {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = sheet.getDataRange().getValues();
    
    var maxNumber = 0;
    var prefix = 'KD-';
    
    for (var i = 1; i < data.length; i++) {
      var customerNumber = data[i][5]; // Spalte F (Kundennummer)
      
      if (customerNumber && typeof customerNumber === 'string' && customerNumber.includes('KD-')) {
        var numberPart = customerNumber.replace(prefix, '').replace(/[^0-9]/g, '');
        
        if (numberPart) {
          var num = parseInt(numberPart);
          if (!isNaN(num) && num > maxNumber) {
            maxNumber = num;
          }
        }
      }
    }
    
    var nextNumber = maxNumber + 1;
    var paddedNumber = nextNumber.toString().padStart(4, '0');
    var result = prefix + paddedNumber;
    
    return ContentService.createTextOutput(result);
    
  } catch (error) {
    return ContentService.createTextOutput("FEHLER: " + error.toString());
  }
}
🚀 Schritt 3: Als Web-App bereitstellen
  1. Klicken Sie auf "Bereitstellen" → "Neue Bereitstellung"
  2. Bei "Typ" wählen: "Web-App"
  3. Wichtige Einstellungen:
    • Ausführen als: "Ich"
    • Zugriff: "Jeder, auch anonym" ⚠️ Sehr wichtig!
  4. Klicken Sie "Bereitstellen"
  5. Bei der ersten Bereitstellung werden Sie nach Berechtigungen gefragt:
    • Klicken Sie "Berechtigungen überprüfen"
    • Wählen Sie Ihr Google-Konto
    • Klicken Sie "Erweitert" → "Zu [Projektname] wechseln"
    • Klicken Sie "Zulassen"
🔗 Schritt 4: URL kopieren und einfügen
  1. Nach der Bereitstellung erhalten Sie eine Web-App-URL
  2. Sie sieht so aus: https://script.google.com/macros/s/AKfycbx...xyz/exec
  3. Kopieren Sie diese URL komplett!
  4. Fügen Sie sie oben in das Feld "Google Apps Script Webhook URL" ein

Nach der Einrichtung: Erstellen Sie eine Testrechnung. Bei Erfolg sollten die Daten automatisch in Ihrem Google Sheet erscheinen und Sie erhalten die Meldung "PDF erstellt und Daten an Google Sheet gesendet!"

⚠️ Häufige Probleme
  • Zugriff nicht auf "anonym" gesetzt → Neu bereitstellen mit korrekten Einstellungen
  • URL falsch kopiert → Muss mit /exec enden
  • Berechtigungen nicht erteilt → Script im Apps Script Editor manuell ausführen
  • Alte Bereitstellung → Nach Änderungen immer neu bereitstellen
